## 一、设计概述

### 1.1 设计目标

1. **扫描根目录管理**：提供 `local_music_dir` 的完整 CRUD 功能
2. **递归扫描**：支持递归扫描配置目录下的所有音乐文件
3. **性能优化**：批量操作、并发控制、进度追踪
4. **数据一致性**：确保扫描过程中数据的一致性
5. **错误处理**：完善的错误处理和恢复机制

### 1.2 核心流程

```
用户配置扫描目录 (local_music_dir)
    ↓
启动扫描任务
    ↓
读取所有启用的扫描根目录
    ↓
递归扫描每个根目录
    ↓
发现目录 → 存入 music_dir
发现文件 → 解析元数据 → 存入 all_music → 存入 local_music
    ↓
更新扫描统计
```

#### 2.2.1 添加扫描目录

```typescript
/**
 * 添加扫描根目录
 *
 * @param path 目录路径
 * @param displayOrder 显示顺序（可选）
 * @returns 创建的目录记录
 * @throws 如果路径已存在、超过20个限制、路径无效
 */
async function addLocalMusicDir(
  path: string,
  displayOrder?: number
): Promise<LocalMusicDir> {
  // 1. 验证路径
  await validateDirectoryPath(path)

  // 2. 规范化路径
  const normalizedPath = normalizePath(path, process.platform)

  // 3. 检查是否已存在
  const existing = db.prepare('SELECT id FROM local_music_dir WHERE path = ?').get(normalizedPath)
  if (existing) {
    throw new Error('目录已存在')
  }

  // 4. 检查数量限制（最多20个）
  const count = db.prepare('SELECT COUNT(*) as count FROM local_music_dir').get() as { count: number }
  if (count.count >= 20) {
    throw new Error('最多只能添加20个扫描目录')
  }

  // 5. 确定显示顺序
  if (displayOrder === undefined) {
    const maxOrder = db.prepare('SELECT MAX(display_order) as max_order FROM local_music_dir').get() as { max_order: number }
    displayOrder = (maxOrder.max_order ?? -1) + 1
  }

  // 6. 插入记录
  const stmt = db.prepare(`
    INSERT INTO local_music_dir (path, display_order, enabled)
    VALUES (?, ?, 1)
  `)
  const result = stmt.run(normalizedPath, displayOrder)

  // 7. 返回创建的记录
  return getLocalMusicDirById(result.lastInsertRowid as number)
}
```

#### 2.2.2 删除扫描目录

```typescript
/**
 * 删除扫描根目录
 *
 * @param id 目录ID
 * @param options 删除选项
 * @returns 是否删除成功

async function deleteLocalMusicDir(
  id: number,
  options?: {
    removeScannedFiles?: boolean  // 是否同时删除已扫描的文件（默认false）

    throw new Error('目录不存在')

    await removeScannedFilesFromDir(dir.path)
  }

  // 3. 删除目录记录
  const stmt = db.prepare('DELETE FROM local_music_dir WHERE id = ?')
  stmt.run(id)

  return true
}

/**
 * 删除指定目录下已扫描的文件
 */
async function removeScannedFilesFromDir(rootPath: string): Promise<void> {
  // 1. 查找所有相关的 music_dir 记录（路径以 rootPath 开头）
  const normalizedRoot = normalizePath(rootPath, process.platform)
  const dirs = db.prepare(`
    SELECT id FROM music_dir
    WHERE path LIKE ? || '%'
  `).all(normalizedRoot) as Array<{ id: number }>

  const dirIds = dirs.map(d => d.id)

  if (dirIds.length === 0) {
    return
  }

  // 2. 删除相关的 all_music 记录（级联删除会处理列表表）
  const placeholders = dirIds.map(() => '?').join(',')
  db.prepare(`
    DELETE FROM all_music
    WHERE dir_id IN (${placeholders})
  `).run(...dirIds)

  // 3. 删除相关的 music_dir 记录
  db.prepare(`
    DELETE FROM music_dir
    WHERE id IN (${placeholders})
  `).run(...dirIds)
}
```

#### 2.2.3 更新扫描目录

```typescript
/**
 * 更新扫描目录信息
 *
 * @param id 目录ID
 * @param updates 更新字段
 * @returns 更新后的目录记录
 */
async function updateLocalMusicDir(
  id: number,
  updates: {
    path?: string
    display_order?: number
    enabled?: boolean
  }
): Promise<LocalMusicDir> {
  // 1. 检查目录是否存在
  const existing = getLocalMusicDirById(id)
  if (!existing) {
    throw new Error('目录不存在')
  }

  // 2. 如果更新路径，需要验证
  if (updates.path) {
    await validateDirectoryPath(updates.path)
    updates.path = normalizePath(updates.path, process.platform)

    // 检查新路径是否已存在（排除自己）
    const conflict = db.prepare('SELECT id FROM local_music_dir WHERE path = ? AND id != ?').get(updates.path, id)
    if (conflict) {
      throw new Error('目录路径已存在')
    }
  }

  // 3. 构建更新SQL
  const fields: string[] = []
  const values: any[] = []

  if (updates.path !== undefined) {
    fields.push('path = ?')
    values.push(updates.path)
  }
  if (updates.display_order !== undefined) {
    fields.push('display_order = ?')
    values.push(updates.display_order)
  }
  if (updates.enabled !== undefined) {
    fields.push('enabled = ?')
    values.push(updates.enabled ? 1 : 0)
  }

  fields.push('updated_at = CURRENT_TIMESTAMP')
  values.push(id)

  // 4. 执行更新
  const sql = `UPDATE local_music_dir SET ${fields.join(', ')} WHERE id = ?`
  db.prepare(sql).run(...values)

  // 5. 返回更新后的记录
  return getLocalMusicDirById(id)
}
```

#### 2.2.5 验证目录路径

```typescript
/**
 * 验证目录路径是否有效
 *
 * @param path 目录路径
 * @throws 如果路径无效
 */
async function validateDirectoryPath(path: string): Promise<void> {
  const fs = await import('fs/promises')

  // 1. 检查路径是否存在
  try {
    const stat = await fs.stat(path)
    if (!stat.isDirectory()) {
      throw new Error('路径不是目录')
    }
  } catch (error: any) {
    if (error.code === 'ENOENT') {
      throw new Error('目录不存在')
    }
    throw new Error(`无法访问目录: ${error.message}`)
  }

  // 2. 检查是否有读取权限
  try {
    await fs.access(path, fs.constants.R_OK)
  } catch (error: any) {
    throw new Error('目录没有读取权限')
  }

  // 3. 检查路径长度（避免过长的路径）
  if (path.length > 4096) {
    throw new Error('路径过长')
  }
}
```

## 三、扫描逻辑设计

### 3.1 扫描流程

```typescript
/**
 * 扫描所有配置的目录
 *
 * @param options 扫描选项
 * @returns 扫描结果
 */
async function scanAllDirectories(options: ScanOptions): Promise<ScanResult> {
  // 1. 获取所有启用的扫描根目录
  const scanDirs = getEnabledLocalMusicDirs()

  if (scanDirs.length === 0) {
    return {
      success: 0,
      failed: 0,
      corrupted: 0,
      skipped: 0,
      duration: 0,
      errors: []
    }
  }

  // 2. 初始化扫描结果
  const result: ScanResult = {
    success: 0,
    failed: 0,
    corrupted: 0,
    skipped: 0,
    duration: 0,
    errors: []
  }

  const startTime = Date.now()

  // 3. 逐个扫描每个根目录
  for (const scanDir of scanDirs) {
    try {
      const dirResult = await scanDirectory(scanDir.path, {
        ...options,
        onProgress: (progress) => {
          // 更新总体进度
          if (options.onProgress) {
            options.onProgress({
              ...progress,
              currentDirectory: scanDir.path
            })
          }
        }
      })

      // 合并结果
      result.success += dirResult.success
      result.failed += dirResult.failed
      result.corrupted += dirResult.corrupted
      result.skipped += dirResult.skipped
      result.errors.push(...dirResult.errors)
    } catch (error: any) {
      result.errors.push({
        file: scanDir.path,
        error: error.message || '扫描失败'
      })
    }
  }

  result.duration = Date.now() - startTime
  return result
}
```

### 3.2 单目录扫描

```typescript
/**
 * 扫描单个目录（递归）
 *
 * @param rootPath 根目录路径
 * @param options 扫描选项
 * @returns 扫描结果
 */
async function scanDirectory(rootPath: string, options: ScanOptions): Promise<ScanResult> {
  const result: ScanResult = {
    success: 0,
    failed: 0,
    corrupted: 0,
    skipped: 0,
    duration: 0,
    errors: []
  }

  const startTime = Date.now()

  // 1. 规范化根目录路径
  const normalizedRoot = normalizePath(rootPath, process.platform)

  // 2. 收集所有音乐文件（递归）
  const files = await collectMusicFiles(normalizedRoot, options)

  // 3. 批量获取目录ID映射（性能优化）
  const dirPaths = [...new Set(files.map(f => parsePath(f).dirPath))]
  const dirIdMap = await batchGetOrCreateMusicDir(db, dirPaths)

  // 4. 处理文件（并发）
  const tasks = files.map(file => async () => {
    try {
      const processed = await processMusicFile(file, dirIdMap, options)
      if (processed.success) {
        result.success++
      } else if (processed.corrupted) {
        result.corrupted++
      } else {
        result.skipped++
      }
    } catch (error: any) {
      result.failed++
      result.errors.push({
        file,
        error: error.message || '处理失败'
      })
    }
  })

  // 5. 并发执行（控制并发数）
  await executeWithConcurrency(tasks, options.concurrency || 10)

  result.duration = Date.now() - startTime
  return result
}
```

### 3.3 文件收集

```typescript
/**
 * 递归收集所有音乐文件
 *
 * @param dir 目录路径
 * @param options 扫描选项
 * @param files 文件列表（递归使用）
 * @returns 文件路径数组
 */
async function collectMusicFiles(
  dir: string,
  options: ScanOptions,
  files: string[] = []
): Promise<string[]> {
  const fs = await import('fs/promises')

  try {
    const entries = await fs.readdir(dir, { withFileTypes: true })

    for (const entry of entries) {
      const fullPath = join(dir, entry.name)

      // 检查排除路径
      if (options.excludePaths?.some(exclude => fullPath.includes(exclude))) {
        continue
      }

      if (entry.isDirectory()) {
        // 递归扫描子目录
        await collectMusicFiles(fullPath, options, files)
      } else if (entry.isFile()) {
        // 检查文件扩展名
        const ext = extname(entry.name).toLowerCase()
        if (options.fileTypes?.includes(ext)) {
          files.push(fullPath)
        }
      }
    }
  } catch (error: any) {
    // 忽略权限错误等，记录但不中断扫描
    console.warn(`无法读取目录 ${dir}:`, error.message)
  }

  return files
}
```

### 3.4 文件处理

```typescript
/**
 * 处理单个音乐文件
 *
 * @param filePath 文件路径
 * @param dirIdMap 目录ID映射
 * @param options 扫描选项
 * @returns 处理结果
 */
async function processMusicFile(
  filePath: string,
  dirIdMap: Record<string, number>,
  options: ScanOptions
): Promise<{
  success: boolean
  corrupted: boolean
  musicId?: number
}> {
  // 1. 解析路径
  const { dirPath, fileName } = parsePath(filePath)
  const normalizedDirPath = normalizePath(dirPath, process.platform)

  // 2. 获取目录ID
  const dirId = dirIdMap[normalizedDirPath]
  if (!dirId) {
    throw new Error(`目录ID未找到: ${normalizedDirPath}`)
  }

  // 3. 检查文件是否已存在
  const existing = db.prepare(`
    SELECT id FROM all_music
    WHERE dir_id = ? AND file_name = ?
  `).get(dirId, fileName) as { id: number } | undefined

  if (existing) {
    // 文件已存在，检查是否需要更新
    const fileStat = await fs.stat(filePath)
    const existingMusic = db.prepare('SELECT * FROM all_music WHERE id = ?').get(existing.id)

    // 如果文件大小或修改时间变化，可能需要重新扫描
    if (options.forceRescan ||
        existingMusic.file_size !== fileStat.size ||
        existingMusic.updated_at < fileStat.mtime.toISOString()) {
      // 重新扫描
      return await rescanMusicFile(filePath, existing.id, dirId, fileName, options)
    }

    // 确保在 local_music 列表中
    ensureInLocalMusic(existing.id)

    return { success: false, corrupted: false }
  }

  // 4. 检查文件是否存在
  let isExists = true
  try {
    await fs.access(filePath, fs.constants.F_OK)
  } catch {
    isExists = false
  }

  // 5. 计算文件哈希（用于去重）
  const fileHash = await calculateFileHash(filePath)

  // 6. 检测文件是否损坏
  const isCorrupted = await detectCorruptedFile(filePath)
  if (isCorrupted) {
    // 插入损坏文件记录
    insertCorruptedMusic(dirId, fileName, filePath, fileHash, '文件损坏')
    return { success: false, corrupted: true }
  }

  // 7. 解析元数据
  const metadata = await parseMusicMetadata(filePath)

  // 8. 获取文件信息
  const fileStat = await fs.stat(filePath)

  // 9. 插入 all_music 记录
  const musicId = insertMusicRecord({
    dir_id: dirId,
    file_name: fileName,
    title: metadata.title || fileName.replace(/\.[^/.]+$/, ''),
    artist: metadata.artist || '未知艺术家',
    album: metadata.album || null,
    year: metadata.year || null,
    genre: metadata.genre || null,
    file_size: fileStat.size,
    file_hash: fileHash,
    file_extension: extname(fileName).toLowerCase(),
    duration: metadata.duration || 0,
    bitrate: metadata.bitrate || 0,
    sample_rate: metadata.sampleRate || 0,
    channels: metadata.channels || 0,
    cover_path: metadata.coverPath || null,
    lyrics_path: null,
    is_exists: isExists ? 1 : 0,
    is_playable: 1,
    play_error_reason: null
  })

  // 10. 添加到 local_music 列表
  addToLocalMusic(musicId)

  return { success: true, corrupted: false, musicId }
}
```

### 3.5 批量目录ID获取

```typescript
/**
 * 批量获取或创建目录ID（使用事务优化）
 */
async function batchGetOrCreateMusicDir(
  db: Database,
  dirPaths: string[],
  platform: NodeJS.Platform = process.platform
): Promise<Record<string, number>> {
  if (dirPaths.length === 0) {
    return {}
  }

  // 规范化所有路径
  const normalizedPaths = dirPaths.map(path => normalizePath(path, platform))
  const uniquePaths = [...new Set(normalizedPaths)]

  // 批量查询已存在的目录
  const placeholders = uniquePaths.map(() => '?').join(',')
  const selectStmt = db.prepare(`
    SELECT id, path FROM music_dir
    WHERE path IN (${placeholders})
  `)
  const existing = selectStmt.all(...uniquePaths) as Array<{ id: number; path: string }>

  const result: Record<string, number> = {}
  const existingMap = new Map<string, number>()

  // 构建已存在目录的映射
  for (const dir of existing) {
    existingMap.set(dir.path, dir.id)
    result[dir.path] = dir.id
  }

  // 找出需要创建的目录
  const toCreate = uniquePaths.filter(path => !existingMap.has(path))

  if (toCreate.length > 0) {
    // 使用事务批量插入
    const insertStmt = db.prepare('INSERT INTO music_dir (path) VALUES (?)')
    const insertMany = db.transaction((paths: string[]) => {
      const map: Record<string, number> = {}
      for (const path of paths) {
        const result = insertStmt.run(path)
        map[path] = result.lastInsertRowid as number
      }
      return map
    })

    const newMap = insertMany(toCreate)
    Object.assign(result, newMap)
  }

  return result
}
```

### 3.6 并发控制

```typescript
/**
 * 并发执行任务（控制并发数）
 */
async function executeWithConcurrency<T>(
  tasks: Array<() => Promise<T>>,
  concurrency: number = 10
): Promise<T[]> {
  const results: T[] = []
  let index = 0
  let activeTasks = 0

  const runNext = async (): Promise<void> => {
    while (index < tasks.length) {
      const currentIndex = index++
      activeTasks++

      try {
        const result = await tasks[currentIndex]()
        results[currentIndex] = result

        // 每处理几个文件后让出控制权
        if (currentIndex % 5 === 0) {
          await new Promise(resolve => setImmediate(resolve))
        }
      } catch (error) {
        // 错误已在任务中处理
      } finally {
        activeTasks--
      }
    }
  }

  // 启动并发任务
  const workers = Array(Math.min(concurrency, tasks.length))
    .fill(0)
    .map(() => runNext())

  await Promise.all(workers)
  return results
}
```

### 4.2 扫描管理器

```typescript
class ScanManager {
  private state: ScanState = {
    isScanning: false,
    isPaused: false,
    isCancelled: false,
    progress: null,
    currentDirectory: null
  }

  private currentScanner: FileScanner | null = null

  getState(): ScanState {
    return { ...this.state }
  }

  async startScan(options: ScanOptions): Promise<ScanResult> {
    if (this.state.isScanning) {
      throw new Error('扫描已在进行中')
    }

    this.state.isScanning = true
    this.state.isPaused = false
    this.state.isCancelled = false

    try {
      const scanner = new FileScanner(db)
      this.currentScanner = scanner

      const result = await scanner.scanAllDirectories({
        ...options,
        onProgress: (progress) => {
          this.state.progress = progress
          if (options.onProgress) {
            options.onProgress(progress)
          }
        }
      })

      return result
    } finally {
      this.state.isScanning = false
      this.currentScanner = null
      this.state.progress = null
    }
  }

  pause(): void {
    if (this.state.isScanning && !this.state.isPaused) {
      this.state.isPaused = true
      this.currentScanner?.setPaused(true)
    }
  }

  resume(): void {
    if (this.state.isPaused) {
      this.state.isPaused = false
      this.currentScanner?.setPaused(false)
    }
  }

  cancel(): void {
    if (this.state.isScanning) {
      this.state.isCancelled = true
      this.state.isPaused = false
      this.currentScanner?.setCancelled(true)
    }
  }
}
```

## 五、性能优化策略

### 5.1 批量操作

1. **批量获取目录ID**：一次性获取所有需要的目录ID，减少数据库查询
2. **批量插入文件**：使用事务批量插入文件记录
3. **批量更新**：合并多个更新操作到单个事务

### 5.2 缓存策略

1. **目录ID缓存**：缓存 `path -> dir_id` 映射
2. **文件哈希缓存**：缓存已计算的文件哈希
3. **元数据缓存**：缓存已解析的元数据（可选）

### 5.3 并发控制

1. **文件处理并发**：控制同时处理的文件数量（默认10个）
2. **I/O 操作异步化**：所有文件系统操作使用异步API
3. **进度更新节流**：限制进度更新频率（每100ms一次）

### 5.4 数据库优化

1. **索引优化**：确保关键查询字段都有索引
2. **事务使用**：批量操作使用事务
3. **延迟索引更新**：大批量插入时临时禁用索引（SQLite 不支持，但可以批量插入）

## 六、错误处理

### 6.1 错误分类

1. **文件级错误**：单个文件处理失败，不影响整体扫描
2. **目录级错误**：目录访问失败，跳过该目录
3. **系统级错误**：数据库错误等，需要中断扫描

### 6.2 错误恢复

1. **重试机制**：临时错误自动重试
2. **错误记录**：记录所有错误到结果中
3. **部分成功**：即使部分文件失败，也返回成功处理的文件
